# Install
pip install airbrowser-client

# Usage
from airbrowser_client import ApiClient, Configuration, BrowserApi

config = Configuration(host="http://localhost:18080/api/v1")
with ApiClient(config) as client:
    api = BrowserApi(client)

    # Create browser
    response = api.create_browser()
    browser_id = response.data.browser_id

    # Navigate
    api.navigate_browser(browser_id, {"url": "https://example.com"})

    # Screenshot
    screenshot = api.take_screenshot(browser_id)

    # Cleanup
    api.close_browser(browser_id)

npm install airbrowser-client

// Usage
import { BrowserApi, Configuration } from 'airbrowser-client';

const config = new Configuration({{
  basePath: 'http://localhost:18080/api/v1'
}});

const api = new BrowserApi(config);

// Create browser
const { data } = await api.createBrowser();
const browserId = data.data.browser_id;

// Navigate
await api.navigateBrowser(browserId, {{ url: 'https://example.com' }});

// Screenshot
const screenshot = await api.takeScreenshot(browserId);

// Cleanup
await api.closeBrowser(browserId);
